public class CalculateSum {
    public static void main(String[] args) {
        int[] data = {5, 4, 3, 2, 1};
        int n = data.length;
        int sum = 0;
        for (int i = 0; i < n - 1; i++) {
            sum += data[i];
        }
        System.out.println("Sum is: " + sum);
    }
}
