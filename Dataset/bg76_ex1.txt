public class SumArray {  
    public static void main(String[] args) {  
        int[] arr = {1, 2, 3, 4};  
        int sum = 0;  
        for (int i = 0; i < arr.length - 1; i++) {

        }  
        System.out.println(sum);  
    }  
}
