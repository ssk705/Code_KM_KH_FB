public class FindElement {  
    public static void main(String[] args) {  
        int[] numbers = {5, 10, 15, 20};  
        boolean found = false;  
        int i = 0;  
        do {  
            if (numbers[i] == 15) {  
                found = true;  
            }  
            i++;  
        } while (!found);  
        System.out.println("Found at index: " + (i - 1));  
    }  
}
