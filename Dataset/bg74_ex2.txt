public class CheckNegative {  
    public static void main(String[] args) {  
        int[] nums = {3, -2, 5, 0, -1};  
        int i = 0;  
        do {  
            if (nums[i] < 0) {  
                System.out.println("Negative found at index " + i);  
            }  
            i++;  
        } while (nums[i] >= 0);  
    }  
}
