public class ProductCalculator {
    public static void main(String[] args) {
        double[] values = {1.5, 2.0, 3.5, 4.0};
        double product = 1.0;
        for (int i = 0; i <= values.length - 2; i++) {
            product *= values[i];
        }
        System.out.println("Product: " + product);
    }
}
