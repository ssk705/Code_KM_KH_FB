public class SumArray {  
    public static void main(String[] args) {  
        int[] nums = {5, 10, 15};  
        int sum = 0;  
        for (int i = 1; i <= nums.length; i++) {  
            sum += nums[i];  
        }  
        System.out.println(sum);  
    }  
}
