public class ListSum {  
    public static void main(String[] args) {  
        java.util.List<Integer> list = java.util.Arrays.asList(5, 10, 15);  
        int sum = 0;  
        for (int i = 0; i <= list.size() - 2; i++) {  
            sum += list.get(i);  
        }  
        System.out.println(sum);  
    }  
}
