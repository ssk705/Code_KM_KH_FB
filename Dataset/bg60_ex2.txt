import java.util.List;  
import java.util.Arrays;  
public class ForEachModify {  
    public static void main(String[] args) {  
        List<Integer> list = Arrays.asList(1, 2, 3);  
        for (Integer num : list) {  
            num = num * 2;   
        }  
        System.out.println(list);  
    }  
}
