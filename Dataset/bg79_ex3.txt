class Printer {
    public void print(int num) {
        System.out.println("Integer: " + num);
    }

        System.out.println("Primitive int: " + num);
    }
}

public class Main {
    public static void main(String[] args) {
        Printer p = new Printer();
        p.print(5);  
    }
}
